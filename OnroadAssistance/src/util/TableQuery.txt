CREATE TABLE CUSTOMERREGISTER 
   (	CUSTOMERNAME VARCHAR2(50), 
	PASSWORD VARCHAR2(50), 
	GENDER NUMBER(1,0), 
	DOB VARCHAR2(20), 
	CONTACTNUMBER NUMBER(19,0), 
	EMAIL VARCHAR2(30) primary key, 
	CUST_ID NUMBER(10,0)
   );


 CREATE TABLE CUSTOMERREQUEST 
   (	CUSTOMERID NUMBER(10,0), 
	CONTACTNUMBER VARCHAR2(19), 
	EMAIL VARCHAR2(30), 
	LOCATION VARCHAR2(30), 
	LATTITUDE NUMBER(10,5), 
	LONGITUDE NUMBER(10,5), 
	MECH_ID NUMBER(10,0), 
	REQUESTID NUMBER primary key, 
	STATUS VARCHAR2(20)
   );


  CREATE TABLE MECHANIC1 
   (	MECH_NAME VARCHAR2(20 ), 
	MECH_PASSWORD VARCHAR2(20 ), 
	MECH_GENDER NUMBER, 
	MECH_DOB VARCHAR2(20 ), 
	MECH_CONTACTNO VARCHAR2(20 ), 
	MECH_EMAILID VARCHAR2(20 ), 
	MECH_LATITUDE NUMBER(20,10), 
	MECH_LONGITUDE NUMBER(20,10), 
	MECH_ID NUMBER
   );


  CREATE TABLE RATING 
   (	MECH_ID NUMBER, 
	CUST_ID NUMBER, 
	RATING NUMBER
   );
